```cpp
#include <bits/stdc++.h>
using namespace std;

int main(){

int arr[9] , total , m ,n;

for (int i = 0 ; i<9; i++){
    cin >> arr[i];
    total = total + arr[i];
}

sort(arr,arr+9);

for(int i = 0 ; i < 9; i++){
    for(int j = i + 1 ; j < 9 ; j++){
        if( total - 100 ==  arr[i] + arr[j]){
            m = i ;
            n = j ;
        }
    }
}
cout <<  "\n";
for(int i = 0 ; i < 9 ; i++){
    if(i != m && i != n){
        cout << arr[i] << "\n";
    }
}
    return 0;
}
```
