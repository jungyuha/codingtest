#### 내 코드

```cpp
#include <bits/stdc++.h>
using namespace std;

int a,b,res;
vector<int> aLst;
int main(){
    cin >> a;
    for(int i=0;i<a;i++){
        cin >> b;
        aLst.push_back(b);
    }

    for(int i=0;i<a;i++){
        int cur = aLst[i],mid_max=-1;
        for(int j=i+1;j<a;j++){
            int next = aLst[j];
            if(mid_max==-1 || ( max(mid_max,next)==next)){
                res++;
                mid_max=next;
            }
            if(cur < next) {break;}
        }
    }

    cout << res << "\n";
    return 0;
}
```

#### 스택 이용 코드

```cpp
#include <bits/stdc++.h>
using namespace std;
typedef long long ll;  
ll n, ret, temp;
stack<pair<ll, ll>> s; 
int main() {
	cin >> n;  
	for(int i = 0; i < n; i++){
		cin >> temp;
		int cnt = 1; 
		while(s.size() && s.top().first <= temp){
			ret += s.top().second; 
			if(s.top().first == temp){ 
				cnt = s.top().second + 1;  
			}else{  
				cnt = 1; 
			}
			s.pop(); 
		}
		if(s.size()) ret++;
        cout << "s.push :: " << temp <<","<<cnt<<"\n";
		s.push({temp, cnt});
	}
	cout << ret << '\n';
    return 0;
} 
```
