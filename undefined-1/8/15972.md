#include <bits/stdc++.h>
using namespace std;

const int INF = 987654321;
int n,m,h,val_[1010][1010];
map <pair<int,int>,int> hor,ver;
priority_queue <tuple<int,int,int>> q1; // 0 가로 , 1 세로
void init(){
    cin >> n >> m >> h; //2 3 5
   fill(val_,val_+1010*1010,h);

   int a;
    for(int i=0;i<=n;i++){
        for(int j=0;j<m;j++){
            cin >> a;
            if(a==-1) continue;
            hor[{i,j}]=a;
            q1.push({0,i,j});
        }
    }
    for(int i=0;i<=m;i++){
        for(int j=0;j<n;j++){
            cin >> a;
            if(a==-1) continue;
            ver[{i,j}]=a;
            q1.push({1,i,j});
        }
    }
}
int main(){
    init();

    while(!q1.empty()){
        int type_= get<0>(q1.top());
        int i = get<1>(q1.top()); int j = get<2>(q1.top());
        int h,cx1,cx2,cy1,cy2;
        if(type_==0){ h = hor[{i,j}]; cx1 = j; cx2 = j ; cy1 = i-1; cy2 = i;
            if(cx1 < 0 || cy1 <0) {val_}
            val_[cy1][cx1]; val_[cy2][cx2];
        }
        else if(type_==1){ h = ver[{i,j}]; cx1 = i-1; cx2 = i ; cy1 = j; cy2 = j;}
    }
    
}
