#include <bits/stdc++.h>
using namespace std;
typedef long long ll;

const int dx[]={0,1,0,-1};
const int dy[]={-1,0,1,0};
int n,m,mp[55][55],vis[55][55],dp[55][55];

int go(int cx,int cy){
    if(cx<0||cx>=m||cy<0||cy>=n||mp[cy][cx]=='H') return 0;
    if(dp[cy][cx] != -1 ) return dp[cy][cx];
    if(vis[cy][cx]==1 && dp[cy][cx]==-1) return 987654321;
    int res=-987654321;
    vis[cy][cx] = 1;
    for(int i=0;i<4;i++){
        int nx = cx + dx[i];
        int ny = cy + dy[i];
        res=max(res,go(nx,ny)+1);
    }
    dp[cy][cx]=res;
    vis[cy][cx] = 0;
    return res;
}
int main() {
    cin >> n >> m;

    for(int i=0;i<n;i++){
        for(int j=0;j<m;j++){
            cin >> mp[i][j];
        }
    }
    return 0;
}
