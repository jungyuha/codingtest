#include <bits/stdc++.h>
using namespace std;
typedef long long ll;
int t,p,n,mp[10000][4],res[10000];
int main() {

    mp[0][1] =1; mp[0][2] =1; mp[0][3] =1;
    for(int i=1;i<=3;i++){
        for(int j=1;j<=10000;j++){
            if(j-i>=0){
                mp[j][i]+=mp[j-i][i];
                if(j==4){
                    cout << "mp["<<j<<"]["<<i<<"] ="<<mp[j][i]<< ",";
                    cout << "before : mp["<<j-i<<"]["<<i<<"] ="<<mp[j-i][i]<< ",";
                }
                res[j]+=res[j-1];
            }
            
        }
    }

    cin >> t;
    while(t){
        cin >> p;
        cout << mp[4][1] << endl;
        cout << mp[4][2] << endl;
        cout << mp[4][3] << endl;
        t--;
    }

    return 0;
}
